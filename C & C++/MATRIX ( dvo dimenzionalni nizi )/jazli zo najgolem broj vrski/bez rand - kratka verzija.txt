#include <iostream>
using namespace std;

int main()
{
    int n, m, naj=0, br=0;
    cin>>n>>m; int A[n][m];

    for ( int i=0; i<n; i++) { for(int j=0; j<m; j++){ cin>>A[i][j]; } } cout<<endl;

    for ( int i=0; i<n; i++) {
        for(int j=0; j<m; j++){ br+=A[i][j]; }
        if ( br > naj ) { naj = br; }
        if( br !=0 ) { br=0; } }

    for ( int i=0; i<n; i++) {
        for(int j=0; j<m; j++){ br+=A[i][j]; }
            if( br !=0 ) { if ( br == naj )
            { cout<<"i:"<<i<<" | "<<br<<endl; } br=0; } }
}
