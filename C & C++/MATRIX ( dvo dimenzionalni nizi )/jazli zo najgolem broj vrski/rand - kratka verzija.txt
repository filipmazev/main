#include <iostream>
using namespace std;

int main()
{
    int n, m, k, r, naj=0, br=0;
    cin>>n>>m; int A[n][m];

    for ( int i=0; i<n; i++) { for(int j=0; j<m; j++){ A[i][j] = 0; } }

    for ( int i=1; i<=n*2; i++)
    { k = rand() % n; r = rand() % n;
            if ( k > n-1 ){ while ( k > n-1 ){ k = rand() % n; } }
            if ( r > n-1 ){ while ( r > n-1 ){ r = rand() % n; } }
    A[k][r] = 1; } cout<<endl;

    for ( int i=0; i<n; i++) {
        for(int j=0; j<m; j++){ cout<<A[i][j]<<" "; }
      cout<<endl; } cout<<endl;

    for ( int i=0; i<n; i++) {
        for(int j=0; j<m; j++){ br+=A[i][j]; }
            if ( br > naj ) { naj = br; }
            if( br !=0 ) { br=0; } }

    for ( int i=0; i<n; i++) {
        for(int j=0; j<m; j++){ br+=A[i][j]; }
            if( br !=0 ) { if ( br == naj )
            { cout<<"i:"<<i<<" | "<<br<<endl; } br=0; } }
}
