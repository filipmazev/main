#include <iostream>
using namespace std;
int main()
{
int m=3, n=3;
int m2=3, n2=3;
int A[m][n];
int C[3][3];

//za A
cout<<" Vnesi prva matrica: "<<endl;
for(int i=0; i<m; i++)
{
	for(int j=0; j<n; j++)
	{
		cin>>A[i][j];
	}
}

for(int i=0; i<m; i++)
{
	for(int j=0; j<n; j++)
	{
		cout<<A[i][j]<<" ";
	}
cout<<endl;
}

cout<<" Vnesi broj po koj sakash da mnozhish so prvata matrica: "<<endl;
int br;
cin>>br;
int D[m2][n2];
for(int i=0; i<m2; i++)
{
	for(int j=0; j<n2; j++)
	{
	D[i][j]=br;
	}
}
for(int i=0; i<m2; i++)
{
	for(int j=0; j<n2; j++)
	{
		cout<<D[i][j]<<" ";
	}
cout<<endl;
}

for(int i=0; i<m; ++i)
{
	for(int j=0; j<n2; ++j)
	{
		for(int k=0; k<n; ++k)
		{
			C[i][j]+=A[i][k] * D[k][j];
		}
	}
}
for(int i=0; i<m; ++i)
{
	for(int j=0; j<n2; ++j)
	{
		cout<<C[i][j]<<" ";
	}
cout<<endl;
}

return 0;
}
