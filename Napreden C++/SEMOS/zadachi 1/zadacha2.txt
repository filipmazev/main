#include <iostream>
using namespace std;
int main()
{
int A[3][2]={{1, 2},{2, 3},{1, 2}};
int B[3][2]={{0, 1},{2, -1},{-1, 2}};
int C[3][2];

for(int i=0; i<3; i++)
{
	for(int j=0; j<2; j++)
	{
		C[i][j]=A[i][j]-B[i][j];
	}
}

for(int i=0; i<3; i++)
{
	for(int j=0; j<2; j++)
	{	
		cout<<C[i][j];
	}
cout<<endl;
}
	
return 0;
}
